#version 330 core

in float lightVal;
in vec3 Color;
in vec2 TexCoord;
out vec4 FragColor;

uniform sampler2D blockTexture;

void main(){
	vec4 TexturePixel = texture(blockTexture, TexCoord);
	FragColor = vec4(TexturePixel.x * Color.x, TexturePixel.y * Color.y, TexturePixel.z * Color.z, TexturePixel.w);
}
