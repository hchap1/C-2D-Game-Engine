#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec2 aTexCoord;
layout (location = 2) in float blockID;

out vec3 Color;
out vec2 TexCoord;

uniform float cameraX;
uniform float cameraY;

void main(){
	if (blockID < 4) {
		gl_Position = vec4(aPos.x + cameraX, aPos.y + cameraY, aPos.z, 1.0);
	}
	else {
		gl_Position = vec4(aPos, 1.0);
	}
	TexCoord = vec2(aTexCoord.x, (aTexCoord.y / 8) + (0.875 - (blockID / 8)));
}
