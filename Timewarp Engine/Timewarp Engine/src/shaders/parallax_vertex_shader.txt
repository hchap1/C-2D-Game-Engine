#version 330 core
layout (location = 0) in vec2 aPos;
layout (location = 1) in vec2 aTexCoord;
layout (location = 2) in float backgroundID;

out vec2 TexCoord;
uniform float size;
uniform float cameraX;
uniform float cameraY;

void main() {
	vec2 shiftedPosition = vec2(aPos.x + (cameraX / (6 - backgroundID * 4)), aPos.y + (cameraY / (6 - backgroundID * 4)));
	gl_Position = vec4(shiftedPosition, 1.0, 1.0);
	TexCoord = vec2(aTexCoord.x, aTexCoord.y);
}
